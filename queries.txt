mutation addUser{
  add_user{
    id
    user_id
    created_ts
    modified_ts
  }
}

mutation addPost($input: addPostInput!){
  add_post(input:$input){
    post{
      id
      title
      post_content
    }
  }
}

mutation addComment($input: addCommentInput!){
  add_comment(input:$input){
    post{
      id
      user
      title
      post_content
      image_url
      comments{
        edges{
          cursor
          node{
            id
            user
            post
            comment_content
            image_url
          }
        }
      }
    }
    comment{
      id
      user
      post
      comment_content
      image_url
    }
  }
}
